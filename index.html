Chapter I

Enable Macros

Before any VBA code can run, PowerPoint needs the Developer tab and correct Trust Center settings.

Show the Developer tab

File → Options → Customize Ribbon → check Developer → OK

Set macro security

File → Options → Trust Center → Trust Center Settings → Macro Settings → choose Disable VBA macros with notification

Save as .pptm

File → Save As → change type to PowerPoint Macro-Enabled Presentation (.pptm)

Open the VBA Editor

Press Alt+F11 or Developer → Visual Basic. Insert → Module to create a new module.

⚠

Important: Never set Trust Center to "Enable all macros" permanently. Use "Disable with notification" and approve files individually.

    ' ===== END YOUR CODE =====

    Exit Sub

    MsgBox "Error " & Err.Number & ": " & Err.Description, vbCritical

Text & Fonts

Replace fonts, fix sizing, standardize text across every slide in seconds instead of hours.

✎ Find & Replace Font Family

~15 min/deck
EssentialAgency

Replaces every instance of one font with another across all slides, including tables, groups, and text boxes.

VBA
Sub FindReplaceFont()
    On Error GoTo ErrHandler
    Application.ScreenUpdating = False

    Dim oldFont As String, newFont As String
    oldFont = InputBox("Current font name:", "Find Font", "Arial")
    If oldFont = "" Then Exit Sub
    newFont = InputBox("Replace with:", "Replace Font", "Calibri")
    If newFont = "" Then Exit Sub

    Dim sld As Slide, shp As Shape, count As Long
    For Each sld In ActivePresentation.Slides
        For Each shp In sld.Shapes
            count = count + ReplaceShapeFont(shp, oldFont, newFont)
        Next shp
    Next sld

    Application.ScreenUpdating = True
    MsgBox "Replaced " & count & " text runs.", vbInformation
    Exit Sub

ErrHandler:
    Application.ScreenUpdating = True
    MsgBox "Error: " & Err.Description, vbCritical
End Sub

Private Function ReplaceShapeFont(shp As Shape, oldF As String, newF As String) As Long
    Dim c As Long
    If shp.HasTextFrame Then
        If shp.TextFrame.HasText Then
            Dim r As TextRange2
            For Each r In shp.TextFrame2.TextRange.Runs
                If LCase(r.Font.Name) = LCase(oldF) Then
                    r.Font.Name = newF
                    c = c + 1
                End If
            Next r
        End If
    End If
    If shp.HasTable Then
        Dim row As Long, col As Long
        For row = 1 To shp.Table.Rows.Count
            For col = 1 To shp.Table.Columns.Count
                Dim cell As Shape
                Set cell = shp.Table.Cell(row, col).Shape
                c = c + ReplaceShapeFont(cell, oldF, newF)
            Next col
        Next row
    End If
    If shp.Type = msoGroup Then
        Dim grp As Shape
        For Each grp In shp.GroupItems
            c = c + ReplaceShapeFont(grp, oldF, newF)
        Next grp
    End If
    ReplaceShapeFont = c
End Function

🔧 Standardize Font Size

~5 min/deck
Beginner

Sets all body text to a target size while preserving titles. Useful for ensuring consistency after a multi-author edit.

VBA
Sub StandardizeFontSize()
    On Error GoTo ErrHandler
    Application.ScreenUpdating = False

    Dim targetSize As String
    targetSize = InputBox("Set body text to size (pt):", "Font Size", "12")
    If targetSize = "" Then Exit Sub
    Dim sz As Single
    sz = CSng(targetSize)

    Dim sld As Slide, shp As Shape, count As Long
    For Each sld In ActivePresentation.Slides
        For Each shp In sld.Shapes
            If shp.HasTextFrame Then
                If shp.TextFrame.HasText Then
                    ' Skip titles (placeholder type 1 = title)
                    If shp.PlaceholderFormat.Type <> ppPlaceholderTitle And _
                       shp.PlaceholderFormat.Type <> ppPlaceholderCenterTitle Then
                        shp.TextFrame.TextRange.Font.Size = sz
                        count = count + 1
                    End If
                End If
            End If
        Next shp
    Next sld

    Application.ScreenUpdating = True
    MsgBox "Updated " & count & " shapes to " & sz & "pt.", vbInformation
    Exit Sub
ErrHandler:
    Application.ScreenUpdating = True
    If Err.Number = 91 Then Resume Next ' shape not a placeholder
    MsgBox "Error: " & Err.Description, vbCritical
End Sub

🔍 Find & Replace Text Across All Slides

~10 min/deck
EssentialBeginner

Global find and replace that works in text boxes, tables, and notes. PowerPoint's built-in find/replace misses many shapes.

VBA
Sub GlobalFindReplace()
    On Error GoTo ErrHandler
    Application.ScreenUpdating = False

    Dim findStr As String, replStr As String
    findStr = InputBox("Find what:", "Find & Replace")
    If findStr = "" Then Exit Sub
    replStr = InputBox("Replace with:", "Find & Replace", "")

    Dim sld As Slide, shp As Shape, count As Long
    For Each sld In ActivePresentation.Slides
        For Each shp In sld.Shapes
            count = count + ReplaceInShape(shp, findStr, replStr)
        Next shp
        ' Also check notes page
        If sld.HasNotesPage Then
            Dim ns As Shape
            For Each ns In sld.NotesPage.Shapes
                count = count + ReplaceInShape(ns, findStr, replStr)
            Next ns
        End If
    Next sld

    Application.ScreenUpdating = True
    MsgBox "Made " & count & " replacements.", vbInformation
    Exit Sub
ErrHandler:
    Application.ScreenUpdating = True
    MsgBox "Error: " & Err.Description, vbCritical
End Sub

Private Function ReplaceInShape(shp As Shape, f As String, r As String) As Long
    Dim c As Long
    If shp.HasTextFrame Then
        If shp.TextFrame.HasText Then
            Dim tr As TextRange
            Set tr = shp.TextFrame.TextRange.Replace(f, r)
            Do While Not tr Is Nothing
                c = c + 1
                Set tr = shp.TextFrame.TextRange.Replace(f, r)
            Loop
        End If
    End If
    If shp.HasTable Then
        Dim rw As Long, cl As Long
        For rw = 1 To shp.Table.Rows.Count
            For cl = 1 To shp.Table.Columns.Count
                c = c + ReplaceInShape(shp.Table.Cell(rw, cl).Shape, f, r)
            Next cl
        Next rw
    End If
    If shp.Type = msoGroup Then
        Dim gs As Shape
        For Each gs In shp.GroupItems
            c = c + ReplaceInShape(gs, f, r)
        Next gs
    End If
    ReplaceInShape = c
End Function

📷 Resize All Images to Fit Slide

~10 min/deck
EssentialIntermediate

Scales all pictures to fit within slide bounds while maintaining aspect ratio. Centers them on the slide.

VBA
Sub ResizeAllImagesToFit()
    On Error GoTo ErrHandler
    Application.ScreenUpdating = False

    Dim sld As Slide, shp As Shape, count As Long
    Dim slideW As Single, slideH As Single
    slideW = ActivePresentation.PageSetup.SlideWidth
    slideH = ActivePresentation.PageSetup.SlideHeight

    For Each sld In ActivePresentation.Slides
        For Each shp In sld.Shapes
            If shp.Type = msoPicture Or shp.Type = msoLinkedPicture Then
                Dim ratio As Single
                ratio = shp.Width / shp.Height

                If shp.Width > slideW Or shp.Height > slideH Then
                    If slideW / slideH > ratio Then
                        shp.Height = slideH
                        shp.Width = slideH * ratio
                    Else
                        shp.Width = slideW
                        shp.Height = slideW / ratio
                    End If
                End If

                ' Center on slide
                shp.Left = (slideW - shp.Width) / 2
                shp.Top = (slideH - shp.Height) / 2
                count = count + 1
            End If
        Next shp
    Next sld

    Application.ScreenUpdating = True
    MsgBox "Resized & centered " & count & " images.", vbInformation
    Exit Sub
ErrHandler:
    Application.ScreenUpdating = True
    MsgBox "Error: " & Err.Description, vbCritical
End Sub

Colors & Branding

Swap brand colors in a single pass, standardize fills, and enforce brand guidelines across entire decks.

🎨 Global Color Swap

~30 min/deck
EssentialAgency

Replaces one color with another across shape fills, text, and borders. Essential for rebranding.

VBA
Sub GlobalColorSwap()
    On Error GoTo ErrHandler
    Application.ScreenUpdating = False

    ' Define old and new colors (edit these)
    Dim oldColor As Long, newColor As Long
    oldColor = RGB(0, 102, 204)   ' old blue
    newColor = RGB(180, 83, 9)   ' new copper

    Dim sld As Slide, shp As Shape, count As Long
    For Each sld In ActivePresentation.Slides
        For Each shp In sld.Shapes
            count = count + SwapShapeColor(shp, oldColor, newColor)
        Next shp
    Next sld

    Application.ScreenUpdating = True
    MsgBox "Swapped " & count & " color instances.", vbInformation
    Exit Sub
ErrHandler:
    Application.ScreenUpdating = True
    MsgBox "Error: " & Err.Description, vbCritical
End Sub

Private Function SwapShapeColor(shp As Shape, oldC As Long, newC As Long) As Long
    Dim c As Long
    On Error Resume Next
    ' Fill
    If shp.Fill.ForeColor.RGB = oldC Then
        shp.Fill.ForeColor.RGB = newC: c = c + 1
    End If
    ' Line
    If shp.Line.ForeColor.RGB = oldC Then
        shp.Line.ForeColor.RGB = newC: c = c + 1
    End If
    ' Text color
    If shp.HasTextFrame Then
        If shp.TextFrame.HasText Then
            Dim r As TextRange2
            For Each r In shp.TextFrame2.TextRange.Runs
                If r.Font.Fill.ForeColor.RGB = oldC Then
                    r.Font.Fill.ForeColor.RGB = newC: c = c + 1
                End If
            Next r
        End If
    End If
    ' Group recursion
    If shp.Type = msoGroup Then
        Dim gs As Shape
        For Each gs In shp.GroupItems
            c = c + SwapShapeColor(gs, oldC, newC)
        Next gs
    End If
    On Error GoTo 0
    SwapShapeColor = c
End Function

📈 Audit All Colors Used in Deck

~1 hr/deck
AgencyAdvanced

Scans every shape and produces a report of all unique colors in use, with counts and hex values. Perfect for QA before client delivery.

VBA
Sub AuditColors()
    On Error GoTo ErrHandler
    Application.ScreenUpdating = False

    Dim colors As Object
    Set colors = CreateObject("Scripting.Dictionary")

    Dim sld As Slide, shp As Shape
    For Each sld In ActivePresentation.Slides
        For Each shp In sld.Shapes
            CollectColors shp, colors
        Next shp
    Next sld

    ' Build report
    Dim report As String, key As Variant
    report = "COLOR AUDIT REPORT" & vbCrLf & String(40, "-") & vbCrLf
    report = report & colors.Count & " unique colors found:" & vbCrLf & vbCrLf
    For Each key In colors.Keys
        Dim rgb_val As Long
        rgb_val = CLng(key)
        report = report & "#" & Right("000000" & Hex(rgb_val), 6) & _
                 "  RGB(" & (rgb_val Mod 256) & "," & ((rgb_val \ 256) Mod 256) & "," & (rgb_val \ 65536) & ")  x" & colors(key) & vbCrLf
    Next key

    Application.ScreenUpdating = True
    MsgBox report, vbInformation, "Color Audit"
    Exit Sub
ErrHandler:
    Application.ScreenUpdating = True
    MsgBox "Error: " & Err.Description, vbCritical
End Sub

Private Sub CollectColors(shp As Shape, dict As Object)
    On Error Resume Next
    Dim clr As Long
    clr = shp.Fill.ForeColor.RGB
    If Err.Number = 0 Then
        If dict.Exists(CStr(clr)) Then dict(CStr(clr)) = dict(CStr(clr)) + 1 Else dict.Add CStr(clr), 1
    End If
    Err.Clear
    If shp.HasTextFrame Then
        If shp.TextFrame.HasText Then
            clr = shp.TextFrame.TextRange.Font.Color.RGB
            If Err.Number = 0 Then
                If dict.Exists(CStr(clr)) Then dict(CStr(clr)) = dict(CStr(clr)) + 1 Else dict.Add CStr(clr), 1
            End If
            Err.Clear
        End If
    End If
    If shp.Type = msoGroup Then
        Dim gs As Shape
        For Each gs In shp.GroupItems: CollectColors gs, dict: Next gs
    End If
    On Error GoTo 0
End Sub

Chapter V

Layout & Alignment

Align, distribute, and position shapes with pixel precision across slides.

▦ Align Selected Shapes to Grid

~5 min/deck
EssentialBeginner

Distributes selected shapes evenly in a row, aligning their tops and spacing them equally.

VBA
Sub DistributeSelectedHorizontally()
    On Error GoTo ErrHandler

    Dim sel As Selection
    Set sel = ActiveWindow.Selection
    If sel.Type <> ppSelectionShapes Then
        MsgBox "Select 2+ shapes first.", vbExclamation: Exit Sub
    End If
    If sel.ShapeRange.Count < 2 Then
        MsgBox "Need at least 2 shapes.", vbExclamation: Exit Sub
    End If

    Dim sr As ShapeRange
    Set sr = sel.ShapeRange

    ' Align tops
    Dim topY As Single
    topY = sr(1).Top
    Dim i As Long
    For i = 2 To sr.Count
        If sr(i).Top < topY Then topY = sr(i).Top
    Next i

    ' Sort by left position
    Dim order() As Long, n As Long
    n = sr.Count
    ReDim order(1 To n)
    For i = 1 To n: order(i) = i: Next i
    Dim j As Long, tmp As Long
    For i = 1 To n - 1
        For j = i + 1 To n
            If sr(order(j)).Left < sr(order(i)).Left Then
                tmp = order(i): order(i) = order(j): order(j) = tmp
            End If
        Next j
    Next i

    ' Calculate total width and spacing
    Dim totalW As Single
    For i = 1 To n: totalW = totalW + sr(order(i)).Width: Next i
    Dim startX As Single, endX As Single
    startX = sr(order(1)).Left
    endX = sr(order(n)).Left + sr(order(n)).Width
    Dim gap As Single
    gap = (endX - startX - totalW) / (n - 1)

    ' Apply positions
    Dim curX As Single
    curX = startX
    For i = 1 To n
        sr(order(i)).Top = topY
        sr(order(i)).Left = curX
        curX = curX + sr(order(i)).Width + gap
    Next i

    MsgBox "Aligned " & n & " shapes.", vbInformation
    Exit Sub
ErrHandler:
    MsgBox "Error: " & Err.Description, vbCritical
End Sub

📊 Standardize All Chart Formatting

Sub StandardizeCharts()
    On Error GoTo ErrHandler
    Application.ScreenUpdating = False

    Dim sld As Slide, shp As Shape, count As Long
    For Each sld In ActivePresentation.Slides
        For Each shp In sld.Shapes
            If shp.HasChart Then
                With shp.Chart
                    ' Font standardization
                    .ChartArea.Font.Name = "Calibri"
                    .ChartArea.Font.Size = 10
                    .ChartArea.Font.Color = RGB(68, 64, 60)

                    ' Remove chart border
                    .ChartArea.Border.LineStyle = xlLineStyleNone

                    ' Format title if exists
                    If .HasTitle Then
                        .ChartTitle.Font.Size = 12
                        .ChartTitle.Font.Bold = True
                    End If

                    ' Light gridlines
                    On Error Resume Next
                    .Axes(xlValue).MajorGridlines.Format.Line.ForeColor.RGB = RGB(230, 230, 230)
                    .Axes(xlValue).MajorGridlines.Format.Line.Weight = 0.5
                    On Error GoTo ErrHandler
                End With
                count = count + 1
            End If
        Next shp
    Next sld

    Application.ScreenUpdating = True
    MsgBox "Formatted " & count & " charts.", vbInformation
    Exit Sub
ErrHandler:
    Application.ScreenUpdating = True
    MsgBox "Error: " & Err.Description, vbCritical
End Sub

🗑 Remove Unused Slide Masters

Sub RemoveUnusedLayouts()
    On Error GoTo ErrHandler

    Dim pres As Presentation
    Set pres = ActivePresentation
    Dim count As Long

    ' Collect used layout names
    Dim usedLayouts As Object
    Set usedLayouts = CreateObject("Scripting.Dictionary")
    Dim sld As Slide
    For Each sld In pres.Slides
        Dim key As String
        key = sld.CustomLayout.Name & "|" & sld.CustomLayout.Parent.Name
        If Not usedLayouts.Exists(key) Then usedLayouts.Add key, True
    Next sld

    ' Delete unused layouts (iterate backwards)
    Dim dm As Design
    For Each dm In pres.Designs
        Dim i As Long
        For i = dm.SlideMaster.CustomLayouts.Count To 1 Step -1
            key = dm.SlideMaster.CustomLayouts(i).Name & "|" & dm.Name
            If Not usedLayouts.Exists(key) Then
                On Error Resume Next
                dm.SlideMaster.CustomLayouts(i).Delete
                If Err.Number = 0 Then count = count + 1
                Err.Clear
                On Error GoTo ErrHandler
            End If
        Next i
    Next dm

    MsgBox "Removed " & count & " unused layouts.", vbInformation
    Exit Sub
ErrHandler:
    MsgBox "Error: " & Err.Description, vbCritical
End Sub

Sub MergeAllFromFolder()

        If fileName <> ActivePresentation.Name Then

            slideCount = ActivePresentation.Slides.Count - insertAt + slideCount
            fileCount = fileCount + 1
        End If
        fileName = Dir()
    Loop

    MsgBox "Merged " & fileCount & " files (" & ActivePresentation.Slides.Count & " total slides).", vbInformation
    Exit Sub
ErrHandler:
    MsgBox "Error: " & Err.Description, vbCritical

Sub GenerateSlidesFromExcel()
    On Error GoTo ErrHandler
    Application.ScreenUpdating = False

    ' Pick Excel file
    Dim fd As FileDialog
    Set fd = Application.FileDialog(msoFileDialogFilePicker)
    fd.Title = "Select Excel data file"
    fd.Filters.Add "Excel Files", "*.xlsx;*.xls"
    If fd.Show <> -1 Then Exit Sub

    Dim xlApp As Object, xlWB As Object, xlWS As Object
    Set xlApp = CreateObject("Excel.Application")
    xlApp.Visible = False
    Set xlWB = xlApp.Workbooks.Open(fd.SelectedItems(1))
    Set xlWS = xlWB.Sheets(1)

    ' Read headers from row 1
    Dim colCount As Long
    colCount = xlWS.Cells(1, xlWS.Columns.Count).End(-4159).Column ' xlToLeft

    Dim headers() As String
    ReDim headers(1 To colCount)
    Dim col As Long
    For col = 1 To colCount
        headers(col) = CStr(xlWS.Cells(1, col).Value)
    Next col

    ' Template is slide 1
    Dim pres As Presentation
    Set pres = ActivePresentation
    Dim templateSlide As Slide
    Set templateSlide = pres.Slides(1)

    ' Data rows start at 2
    Dim row As Long, lastRow As Long
    lastRow = xlWS.Cells(xlWS.Rows.Count, 1).End(-4162).row ' xlUp

    For row = 2 To lastRow
        templateSlide.Copy
        pres.Slides.Paste pres.Slides.Count + 1
        Dim newSlide As Slide
        Set newSlide = pres.Slides(pres.Slides.Count)

        ' Replace placeholders
        Dim shp As Shape
        For Each shp In newSlide.Shapes
            If shp.HasTextFrame Then
                If shp.TextFrame.HasText Then
                    Dim txt As String
                    txt = shp.TextFrame.TextRange.Text
                    For col = 1 To colCount
                        txt = Replace(txt, "{{" & headers(col) & "}}", CStr(xlWS.Cells(row, col).Value))
                    Next col
                    shp.TextFrame.TextRange.Text = txt
                End If
            End If
        Next shp
    Next row

    ' Delete template slide
    pres.Slides(1).Delete

    xlWB.Close False
    xlApp.Quit
    Set xlApp = Nothing

    Application.ScreenUpdating = True
    MsgBox "Generated " & (lastRow - 1) & " slides from Excel data.", vbInformation
    Exit Sub
ErrHandler:
    Application.ScreenUpdating = True
    On Error Resume Next
    xlWB.Close False: xlApp.Quit
    MsgBox "Error: " & Err.Description, vbCritical
End Sub

Sub ListAllHyperlinks()

    report = "HYPERLINK REPORT" & vbCrLf & String(50, "-") & vbCrLf

    Dim sld As Slide, shp As Shape

                    Next
                End If
            End If
        Next shp
    Next sld

    If count = 0 Then report = report & "No hyperlinks found."
    report = report & vbCrLf & "Total: " & count & " hyperlinks"
    MsgBox report, vbInformation, "Hyperlinks"
    Exit Sub

Sub AutoBackup()
    On Error GoTo ErrHandler

    Dim pres As Presentation
    Set pres = ActivePresentation

    If pres.Path = "" Then
        MsgBox "Save the file first before backing up.", vbExclamation: Exit Sub
    End If

    Dim baseName As String, ext As String, backupPath As String
    baseName = Left(pres.Name, InStrRev(pres.Name, ".") - 1)
    ext = Mid(pres.Name, InStrRev(pres.Name, "."))
    backupPath = pres.Path & "\" & baseName & "_backup_" & Format(Now, "yyyy-mm-dd_HHnnss") & ext

    pres.SaveCopyAs backupPath

    MsgBox "Backup saved:" & vbCrLf & backupPath, vbInformation
    Exit Sub
ErrHandler:
    MsgBox "Backup failed: " & Err.Description, vbCritical
End Sub

For Teams

Team Deployment

Distribute macros to your entire team as an add-in with a custom ribbon.

ⓘ

A .ppam add-in is the standard way to distribute macros to a team. It loads automatically with PowerPoint and can include a custom ribbon tab with buttons for your most-used macros.

Step-by-Step: Create a .ppam Add-In

Collect your macros

Create a blank .pptm file. Press Alt+F11, Insert → Module, and paste all your macros into one or more modules.

Save as .ppam

File → Save As → change type to PowerPoint Add-In (.ppam). Default location is %AppData%\Microsoft\AddIns\.

Load the add-in

Developer → PowerPoint Add-Ins → Add New → browse to your .ppam. Check the box to enable it.

Add Custom Ribbon (optional)

Use the Custom UI Editor to create a customUI.xml file inside the .ppam with buttons that call your macros. Example XML below.

Distribute to team

Copy the .ppam file to a shared network drive. Each team member adds it from Developer → Add-Ins. It will auto-load every time PowerPoint opens.

XML
<!-- customUI.xml for Custom Ribbon Tab -->
<customUI xmlns="http://schemas.microsoft.com/office/2006/01/customui">
  <ribbon>
    <tabs>
      <tab id="MacroAtelier" label="Macro Atelier">
        <group id="grpText" label="Text">
          <button id="btnFontSwap" label="Replace Font"
                  onAction="FindReplaceFont" size="large"
                  imageMso="FontColorPicker"/>
          <button id="btnFindReplace" label="Find & Replace"
                  onAction="GlobalFindReplace" size="large"
                  imageMso="ReplaceDialog"/>
        </group>
        <group id="grpExport" label="Export">
          <button id="btnPDF" label="Export PDF"
                  onAction="ExportToPDF" size="large"
                  imageMso="FileSaveAsPdfOrXps"/>
        </group>
        <group id="grpSafety" label="Safety">
          <button id="btnBackup" label="Backup Now"
                  onAction="AutoBackup" size="large"
                  imageMso="FileSaveAsOtherFormats"/>
        </group>
      </tab>
    </tabs>
  </ribbon>
</customUI>
◈

Quick Access Toolbar: For individual users, right-click any macro in the macro list and choose "Add to Quick Access Toolbar" for one-click access without a custom ribbon.
